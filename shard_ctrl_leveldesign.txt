define exe_set_leveldesign
	use ms_mini_mainland
	use ais_leveldesign

	use egs_mainland
	use gpms_mainland
	use ios_mainland
	use rns_mainland
	use fes_solo
	use ts_std
	use rws_std
	use mos
	cfg NSHost="localhost";
	cfg BSHost="localhost";
	cfg MFSHost="";
	cfg ConvertSaveFilesDirectoryToFullPath=0;
	cfg SaveShardRoot="../save_shard";
	cfg NegFiltersDebug= {"LNET"};
	cfg NegFiltersInfo= {"LNET"};
	cfg NegFiltersWarning= {"AES"};
	cfgFile ../cfg/00_base.cfg
	cfgFile ../cfg/01_domain_leveldesign.cfg
	cfgFile ../cfg/02_shard_type_leveldesign.cfg
	cfgAfter NegFiltersDebug+= {"DEFAULT_CB", "LNET"};
	cfgAfter NegFiltersInfo+= {"LNET"};
	cfgAfter NegFiltersWarning+= {"AES"};
	cfgAfter #optional "../local.cfg"
	cfgAfter MaxXPGainPerPlayer = 30.0;
